const token = localStorage.getItem('jwt_token');
const API = 'http://localhost:3000/api/my-profile';

// Get Profile
const profile = await fetch(API, {
  headers: { 'Authorization': `Bearer ${token}` }
}).then(r => r.json());

// Update Profile
await fetch(API, {
  method: 'PUT',
  headers: {
    'Authorization': `Bearer ${token}`,
    'Content-Type': 'application/json'
  },
  body: JSON.stringify({
    first_name: 'Jane',
    last_name: 'Smith',
    phone: '+1234567890'
  })
}).then(r => r.json());

// Update Email
await fetch(`${API}/email`, {
  method: 'PUT',
  headers: {
    'Authorization': `Bearer ${token}`,
    'Content-Type': 'application/json'
  },
  body: JSON.stringify({ email: 'newemail@example.com' })
}).then(r => r.json());

// Request Password Reset
await fetch(`${API}/request-password-reset`, {
  method: 'POST',
  headers: { 'Authorization': `Bearer ${token}` }
}).then(r => r.json());

// Change Password
await fetch(`${API}/change-password`, {
  method: 'POST',
  headers: {
    'Authorization': `Bearer ${token}`,
    'Content-Type': 'application/json'
  },
  body: JSON.stringify({
    oldPassword: 'OldPass123',
    newPassword: 'NewPass456'
  })
}).then(r => r.json());
